#!/usr/bin/env python
#
# A simple SPIDER image converter.
# Output format determined by extension.

import Image
import sys

if len(sys.argv[1:]) < 2:
    print "Syntax: spiconvert.py spiderfile outfile"
    sys.exit()

filename = sys.argv[1]
outfile  = sys.argv[2]

try:
    im = Image.open(filename).convert2byte()
except:
    im = Image.open(filename)

im.save(outfile)  # Converts according to extension
